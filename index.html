    function calcDoNothing(totalDebt, apr=0.24) {

      const monthlyPayment = totalDebt * 0.025;

      const monthlyRate = apr / 12;

      if (totalDebt <= 0 || monthlyPayment <= totalDebt * monthlyRate) return { monthlyPayment:0, totalDebt, monthsToPayoff:0, apr, totalPayback:0 };

      const months = Math.max(1, Math.round(Math.log(monthlyPayment / (monthlyPayment - totalDebt * monthlyRate)) / Math.log(1 + monthlyRate)));

      return { monthlyPayment, totalDebt, monthsToPayoff:months, apr, totalPayback: monthlyPayment * months };

    }

    function getRoute(totalDebt, stateCode, flags) {

      if (!stateCode) return { route:'', reason:'Select a state to calculate routing.' };

      if (totalDebt > 0 && totalDebt < 7000) {

        if (flags.consumerShield.some(f => f.includes('not accepted') || f.includes('does not service') || f.includes('red / non-sell') || f.includes('requires at least'))) {

          return { route:'NO OPTION', reason:'Consumer Shield would normally handle debts below $7,000, but this deal does not currently meet Consumer Shield guidelines.' };

        }

        return { route:'CONSUMER SHIELD', reason:'Debt below $7,000 routes to Consumer Shield when state and account types qualify.' };

      }

      if (NO_DS_STATES.includes(stateCode)) return { route:'NO OPTION', reason:'No current debt settlement coverage in this state for Level Debt or Verify.' };

      if (VERIFY_ONLY_STATES.includes(stateCode)) return { route:'VERIFY', reason:'Level Debt is unavailable in this state, so Verify is the fallback.' };

      return { route:'LEVEL', reason:'Level Debt is prioritized whenever available due to stronger economics.' };

    }

    function compute() {

      const populated = state.rows.filter(r => numberOrZero(r.amount) > 0);

      const totalDebt = populated.reduce((s,r)=>s+numberOrZero(r.amount),0);

      const consumerShield = getConsumerShieldFlags(populated, totalDebt, state.selectedState);

      const level = getLevelFlags(populated, totalDebt, state.selectedState, state);

      const routeInfo = getRoute(totalDebt, state.selectedState, { consumerShield, level });

      const stateTier = state.selectedState ? getStateTier(state.selectedState) : 'GREEN';

      const feeRate = routeInfo.route === 'LEVEL' ? (stateTier === 'GREEN' ? 0.25 : 0.27) : routeInfo.route === 'VERIFY' ? (stateTier === 'GREEN' ? 0.25 : 0.30) : 0;

      const verifyAdjustedDebt = populated.reduce((s,r)=>{ const manual = r.creditorName === 'Other / Manual Entry' || !!r.manualName; return s + getVerifyPayableBalance(numberOrZero(r.amount), r.debtType, manual); },0);

      const grossCommissionRate = routeInfo.route === 'LEVEL' ? 0.08 : routeInfo.route === 'VERIFY' ? 0.06 : 0;

      const repCommissionRate = grossCommissionRate > 0 ? 0.01 : 0;

      const commissionBase = routeInfo.route === 'VERIFY' ? verifyAdjustedDebt : totalDebt;

      const grossCommission = commissionBase * grossCommissionRate;

      const repCommission = commissionBase * repCommissionRate;

      const fundingTierProfit = commissionBase * Math.max(0, grossCommissionRate - repCommissionRate);

      const rejectedForLevel = populated.filter(r => creditorMatches(getCreditorDisplayName(r), LEVEL_REJECTED_CREDITORS));

      const rejectedForVerify = populated.filter(r => creditorMatches(getCreditorDisplayName(r), VERIFY_REJECTED_CREDITORS));

      const doNothing = calcDoNothing(totalDebt);

      let shortest = null, recommended = null;

      if (routeInfo.route === 'LEVEL') {

        const totalCost = totalDebt * LEVEL_SETTLEMENT_RATE + totalDebt * feeRate;

        shortest = { term:12, totalCost, monthlyPayment: totalCost/12 };

        recommended = { term:getLevelMaxTerm(totalDebt), totalCost, monthlyPayment: totalCost/getLevelMaxTerm(totalDebt) };

      } else if (routeInfo.route === 'VERIFY') {

        const maxTerm = getVerifyMaxTerm(totalDebt);

        if (maxTerm) {

          const totalCost = totalDebt * 0.53 + totalDebt * feeRate + 250;

          const shortTerm = Math.max(12, Math.round(maxTerm * 0.72));

          shortest = { term:shortTerm, totalCost, monthlyPayment:(totalCost/shortTerm)+50 };

          recommended = { term:maxTerm, totalCost, monthlyPayment:(totalCost/maxTerm)+50 };

        }

      } else if (routeInfo.route === 'CONSUMER SHIELD') {

        const term = totalDebt > 8800 ? 36 : 24;

        const monthlyPayment = Math.max(220, totalDebt / term);

        const totalCost = monthlyPayment * term;

        shortest = { term, totalCost, monthlyPayment };

        recommended = { term, totalCost, monthlyPayment };

      }

      const flags = [];

      if (routeInfo.route === 'LEVEL') flags.push(...level);

      if (routeInfo.route === 'VERIFY') {

        if (rejectedForVerify.length) flags.push('Verify rejected creditor detected.');

        if (populated.some(r => r.creditorName === 'Other / Manual Entry' || !!r.manualName)) flags.push('Manual creditor present. Verify chargeback policy may apply if creditor is removed.');

        if (populated.some(r => ['Medical Bill','Credit Union Loan','Auto Repo Deficiency','Judgment'].includes(r.debtType))) flags.push('Verify compensation may be reduced based on debt-type caps.');

      }

      if (routeInfo.route === 'CONSUMER SHIELD') {

        flags.push(...consumerShield);

        flags.push('Consumer Shield support is Monday–Friday, 9am–7pm EST.');

        flags.push('Consumer Shield will not send communications after 8pm in the contact’s local time unless already a client.');

        flags.push('Consumer Shield uses Equifax for credit pull.');

      }

      if (routeInfo.route === 'NO OPTION') flags.push(routeInfo.reason);

      return { populated, totalDebt, verifyAdjustedDebt, routeInfo, stateTier, feeRate, grossCommissionRate, repCommissionRate, grossCommission, repCommission, fundingTierProfit, rejectedForLevel, doNothing, shortest, recommended, flags };

    }

        if (LEVEL_REJECTED_TYPES.includes(row.debtType)) alerts.push({ label:'Debt type not accepted', tone:'red' });

      }

      if (route === 'VERIFY') {

        if (creditorMatches(name, VERIFY_REJECTED_CREDITORS)) alerts.push({ label:'Not accepted by Verify', tone:'red' });

        if (isManual) alerts.push({ label:'Manual creditor', tone:'amber' });

        if (['Medical Bill','Credit Union Loan','Auto Repo Deficiency','Judgment'].includes(row.debtType)) alerts.push({ label:'Verify cap applied', tone:'blue' });

      }

      if (route === 'CONSUMER SHIELD') {

        if (CONSUMER_SHIELD_REJECTED_TYPES.includes(row.debtType)) alerts.push({ label:'Not accepted by Consumer Shield', tone:'red' });

        if (CONSUMER_SHIELD_CONDITIONAL_TYPES.includes(row.debtType)) alerts.push({ label:'Conditional / docs needed', tone:'amber' });

      }

      return alerts;

    }

    function renderRows(route) {

      el('rows').innerHTML = state.rows.map((row, index) => {

        const alerts = rowAlerts(row, route);

        const bg = alerts.some(a=>a.tone==='red') ? 'border-red-300 bg-red-50' : alerts.some(a=>a.tone==='amber') ? 'border-amber-300 bg-amber-50' : alerts.some(a=>a.tone==='blue') ? 'border-sky-300 bg-sky-50' : 'border-slate-100 bg-white';

        return `

          <div class="grid grid-cols-[1.2fr,0.65fr,1fr,0.8fr] gap-3 rounded-xl border p-3 ${bg}">

            <div class="space-y-2">

              <select data-index="${index}" data-key="creditorName">${CREDITOR_OPTIONS.map(o=>`<option ${row.creditorName===o?'selected':''} value="${o}">${o || 'Select creditor'}</option>`).join('')}</select>

              ${row.creditorName === 'Other / Manual Entry' ? `<input class="input" data-index="${index}" data-key="manualName" placeholder="Type creditor name" value="${row.manualName || ''}" />` : ''}

              ${alerts.length ? `<div class="flex flex-wrap gap-2 pt-1">${alerts.map(a=>`<span class="rounded-full border px-2 py-1 text-[11px] font-medium ${a.tone==='red'?'bg-red-100 text-red-700 border-red-200':a.tone==='amber'?'bg-amber-100 text-amber-800 border-amber-200':'bg-sky-100 text-sky-700 border-sky-200'}">${a.label}</span>`).join('')}</div>` : ''}

            </div>

            <input class="input" data-index="${index}" data-key="amount" placeholder="$0" value="${row.amount || ''}" />

            <select data-index="${index}" data-key="debtType">${DEBT_TYPES.map(t=>`<option ${row.debtType===t?'selected':''} value="${t}">${t}</option>`).join('')}</select>

            <select multiple class="conditions-select rounded-xl border border-slate-200 bg-white" data-index="${index}" data-key="conditions">

              <option value="inCollections" ${row.inCollections?'selected':''}>Collections</option>

              <option value="inLitigation" ${row.inLitigation?'selected':''}>Litigation</option>

              <option value="isSecured" ${row.isSecured?'selected':''}>Secured</option>

              <option value="paymentMade" ${row.paymentMade?'selected':''}>Payment made</option>

              <option value="businessUnprofitable" ${row.businessUnprofitable?'selected':''}>Business closed / unprofitable</option>

            </select>

          </div>`;

      }).join('');

      document.querySelectorAll('#rows [data-index]').forEach(node => {

        node.addEventListener('input', onFieldChange);

        node.addEventListener('change', onFieldChange);

      });

    }

    function renderComparison(c) {

      const rows = [

        ['Monthly Payments', currency(c.doNothing.monthlyPayment), c.shortest ? currency(c.shortest.monthlyPayment) : '—', c.recommended ? currency(c.recommended.monthlyPayment) : '—'],

        ['Total Estimated Debt', currency(c.totalDebt), c.shortest ? currency(c.totalDebt) : '—', c.recommended ? currency(c.totalDebt) : '—'],

        ['Months to Payoff', `${c.doNothing.monthsToPayoff} months`, c.shortest ? `${c.shortest.term} months` : '—', c.recommended ? `${c.recommended.term} months` : '—'],

        ['Estimated Int Rate', `${(c.doNothing.apr*100).toFixed(0)}% APR`, c.shortest ? '0%' : '—', c.recommended ? '0%' : '—'],

        ['Estimated Total Payback*', currency(c.doNothing.totalPayback), c.shortest ? currency(c.shortest.totalCost) : '—', c.recommended ? currency(c.recommended.totalCost) : '—'],

        ['Estimated Savings', '—', c.shortest ? `<span class="hyper-green">${currency(Math.max(0, c.doNothing.totalPayback - c.shortest.totalCost))}</span>` : '—', c.recommended ? `<span class="hyper-green">${currency(Math.max(0, c.doNothing.totalPayback - c.recommended.totalCost))}</span>` : '—']

      ];

      el('comparisonTable').innerHTML = rows.map((r, idx)=>`

        <tr class="${idx % 2 === 0 ? 'bg-white' : 'bg-slate-50/50'}">

          <td class="border-b px-4 py-4 text-sm font-medium text-slate-600">${r[0]}</td>

          <td class="border-b px-4 py-4 text-sm font-semibold text-slate-900">${r[1]}</td>

          <td class="border-b px-4 py-4 text-sm font-semibold text-slate-900">${r[2]}</td>

          <td class="border-b px-4 py-4 text-sm font-semibold text-slate-900">${r[3]}</td>

        </tr>`).join('');

    }

      const c = compute();

      el('totalDebt').textContent = currency(c.totalDebt);

      renderRows(c.routeInfo.route);

      renderComparison(c);

      const stateFull = state.selectedState ? STATE_NAMES[state.selectedState] : 'No state selected';

      const tierColor = c.stateTier === 'RED' ? '#DC2626' : '#16A34A';

      const tierBg = c.stateTier === 'RED' ? '#FEF2F2' : '#ECFDF5';

      el('topBadges').innerHTML = `

        <span class="rounded-full px-4 py-2 text-sm text-white" style="background:${routeColor}">${c.routeInfo.route || 'Pending'}</span>

        ${c.routeInfo.route === 'CONSUMER SHIELD' ? '' : `<span class="rounded-full px-4 py-2 text-sm border" style="border-color:${tierColor}; color:${tierColor}; background:${tierBg}">${c.stateTier} State Tier</span>`}

        <div class="rounded-xl bg-slate-50 p-4"><div class="text-xs uppercase tracking-wide text-slate-500">Funding Tier Profit</div><div class="mt-1 text-lg font-semibold">${currency(c.fundingTierProfit)}</div></div>

        <div class="rounded-xl bg-slate-50 p-4"><div class="text-xs uppercase tracking-wide text-slate-500">Level Rejected Creditors</div><div class="mt-1 text-lg font-semibold">${c.routeInfo.route==='LEVEL'?c.rejectedForLevel.length:0}</div></div>

    }

    function onFieldChange(e) {

      const index = Number(e.target.dataset.index);

      const key = e.target.dataset.key;

      if (Number.isInteger(index) && state.rows[index]) {

        if (key === 'conditions') {

          const values = Array.from(e.target.selectedOptions).map(opt => opt.value);

          state.rows[index].inCollections = values.includes('inCollections');

          state.rows[index].inLitigation = values.includes('inLitigation');

          state.rows[index].isSecured = values.includes('isSecured');

          state.rows[index].paymentMade = values.includes('paymentMade');

          state.rows[index].businessUnprofitable = values.includes('businessUnprofitable');

        } else {

          state.rows[index][key] = e.target.value;

        }

      }

      render();

    }

    function buildPdf() {

      const c = compute();

      if (!(c.recommended && state.selectedState && c.totalDebt > 0)) return;

      const { jsPDF } = window.jspdf;

      const doc = new jsPDF();

      let y = 18;

      const lineGap = 7;

      const pageWidth = doc.internal.pageSize.getWidth();

      const addLine = (text, size=11, color=[0,0,0], x=14) => {

        doc.setFontSize(size); doc.setTextColor(color[0], color[1], color[2]);

        const lines = doc.splitTextToSize(String(text), pageWidth - 28); doc.text(lines, x, y); y += lines.length * lineGap;

      };

      doc.setFillColor(0,150,136); doc.rect(0,0,pageWidth,24,'F'); doc.setTextColor(255,255,255); doc.setFontSize(18); doc.text('Funding Tier AI - Deal Router',14,15); y=34;

      addLine('State: ' + STATE_NAMES[state.selectedState],11,[91,94,109]);

      addLine('Recommended Route: ' + c.routeInfo.route,11,[91,94,109]);

      addLine('Routing Reason: ' + c.routeInfo.reason,11,[91,94,109]); y += 2;

      addLine('Deal Summary',14,[0,150,136]);

      addLine('Total Enrolled Debt: ' + currency(c.totalDebt));

      addLine('Gross Commission: ' + (c.grossCommissionRate*100).toFixed(0) + '% • ' + currency(c.grossCommission));

      addLine('Rep Commission: ' + (c.repCommissionRate*100).toFixed(0) + '% • ' + currency(c.repCommission));

      addLine('Funding Tier Profit: ' + currency(c.fundingTierProfit));

      if (c.routeInfo.route === 'VERIFY') addLine('Adjusted Verify Compensation Debt: ' + currency(c.verifyAdjustedDebt));

      y += 2;

      addLine('Comparison Summary',14,[0,150,136]);

      addLine('Do Nothing: ' + currency(c.doNothing.monthlyPayment) + ' / ' + c.doNothing.monthsToPayoff + ' months / ' + currency(c.doNothing.totalPayback));

      if (c.shortest) addLine('Shortest Payoff: ' + currency(c.shortest.monthlyPayment) + ' / ' + c.shortest.term + ' months / ' + currency(c.shortest.totalCost) + ' / Savings ' + currency(Math.max(0, c.doNothing.totalPayback - c.shortest.totalCost)));

      if (c.recommended) addLine('Max Term: ' + currency(c.recommended.monthlyPayment) + ' / ' + c.recommended.term + ' months / ' + currency(c.recommended.totalCost) + ' / Savings ' + currency(Math.max(0, c.doNothing.totalPayback - c.recommended.totalCost)));

      y += 2;

      addLine('Creditor Details',14,[0,150,136]);

      c.populated.forEach((row,idx)=> addLine((idx+1) + '. ' + getCreditorDisplayName(row) + ' — ' + currency(numberOrZero(row.amount)) + ' — ' + row.debtType, 10));

      if (c.flags.length) { y += 2; addLine('Guideline Notes',14,[220,38,38]); c.flags.slice(0,10).forEach(flag => addLine('• ' + flag, 10)); }

      y += 4; addLine('*Estimates are based on average credit card interest and minimum payment assumptions. Actual results may vary.',9,[91,94,109]);

      doc.save('funding-tier-ai-deal-router.pdf');

    }

    function init() {

      el('state').innerHTML = '<option value="">Select state</option>' + STATES.map(([abbr,name])=>'<option value="'+abbr+'">'+name+'</option>').join('');

      ['creditorScore','budgetSurplus'].forEach(id => el(id).addEventListener('input', render));

      el('pdfBtn').addEventListener('click', buildPdf);

    }
